<script>
const consoleEl = document.getElementById("console");
const inputEl = document.getElementById("input");

let stage = "game";

const games = [
  "Crypto Hack",
  "Race",
  "Tower Defense",
  "Cafe",
  "Factory",
  "Fishing Frenzy",
  "Gold Quest"
];

function print(text = "") {
  consoleEl.textContent += text + "\n";
  window.scrollTo(0, document.body.scrollHeight);
}

function showGames() {
  print("Select a game:\n");
  games.forEach((g, i) => print(`${i + 1}. ${g}`));
  print("\nEnter your choice number:");
}

function showOptions() {
  print("\nGold Quest selected!\n");
  print("Choose an option:");
  print("1. + Gold");
  print("2. + Chest Reveal");
  print("\nEnter your choice (1 or 2):");
}

function loading(cb) {
  let dots = 0;
  const interval = setInterval(() => {
    consoleEl.textContent =
      consoleEl.textContent.replace(/Installing modules.*/, "") +
      "Installing modules" + ".".repeat(dots % 4);
    dots++;
  }, 300);

  setTimeout(() => {
    clearInterval(interval);
    print("✓");
    cb();
  }, 2500);
}

async function copyFile(filename) {
  const res = await fetch(filename);
  const text = await res.text();
  await navigator.clipboard.writeText(text);
  print("\n[✓] Content copied to clipboard successfully.");
}

inputEl.addEventListener("keydown", async (e) => {
  if (e.key !== "Enter") return;

  const value = inputEl.value.trim();
  inputEl.value = "";

  if (stage === "game") {
    if (value !== "7") {
      print("❌ Only Gold Quest is supported.");
      return;
    }
    stage = "option";
    showOptions();
  }

  else if (stage === "option") {
    loading(async () => {
      if (value === "1") {
        await copyFile("+gold.txt");
      } else if (value === "2") {
        await copyFile("+chestreveal.txt");
      } else {
        print("❌ Invalid option.");
      }
    });
    stage = "done";
  }
});

// Start
showGames();
</script>
